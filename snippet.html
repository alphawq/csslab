fontface 自定义字体定义


		@font-face{ 
		    font-family:'diyfont'; 
		    src: url('diyfont.eot');
		    src: url('diyfont.eot?#iefix') format('embedded-opentype'),
		    url('diyfont.woff') format('woff'),
		    url('diyfont.ttf') format('truetype'),
		    url('diyfont.svg#diyfont') format('svg');
		}
